水仙花数：即三位数的每个位置的立方的和等于这个数。
那么我们如何找出这些水仙花数呢？

    //C033拼接法
    #include <stdio.h>
    #include <stdlib.h>
    int main()
    {
        int i,j,k;
        for(i=1;i<=9;i++)
            for(j=0;j<=9;j++)
                for(k=0;k<=9;k++)
                    if(i*100+j*10+k==i*i*i+j*j*j+k*k*k)
                        printf("%d\n",i*100+j*10+k);
        system("pause");
        return 0;
    }


    //C034
    #include <stdio.h>
