思维挑战11：摘苹果问题-一维数组
-

[代码资源](https://download.csdn.net/download/Btbsja/12155122)

输入所有苹果的高度以及人手可以举到的高度判断可以摘到几个苹果。

    //C036
    #include <stdio.h>
    #include <stdlib.h>
    int main()
    {
        int h,a[10],i,sum;
        for(i=0;i<=9;i++)
            scanf("%d",&a[i]);
            scanf("%d",&h);
        sum=0;
        for(i=0;i<=9;i++)
        {
            if(h>=a[i])
            sum++;
        }
        printf("%d",sum);
        system("pause");
        return 0;
    }
