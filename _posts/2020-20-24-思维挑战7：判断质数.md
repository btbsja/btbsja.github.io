质数：指大于1的自然数，除了1和自身以外无法被其他数整除。
即，我们只需要判断 a 除以 2 至 a-1 不为 0 那么 a 便为质数。

    //C031
    #include <stdio.h>
    #include <stdlib.h>
    int main()
    {
        int a,i,count=0;
        printf("请输入需要判断的数字：");
        scanf("%d",&a);
        for(i=2;i<=a-1;i++)
        {
            if(a%i==0)
            {
                count++;
                printf("%d ",i);
            }
        }
        if(count==0)
            printf("\n质数\n");
        else
            printf("为其约数\n合数\n");
            system("pause");
        return 0;
    }
